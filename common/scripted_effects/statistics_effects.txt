gdp_global_list_effect = {
	clear_global_variable_list = gdp_ranking_global_list
	ordered_country = {
		limit = {
			is_revolutionary = no
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = gdp
		position = 0
		max = 1
		check_range_bounds = no
		set_global_variable = {
			name = top_gdp_country_gvar
			value = this.gdp
		}
	}

	ordered_country = {
		limit = {
			is_revolutionary = no
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}

		set_local_variable = {
			name = gdp_local_var
			value = 0
		}
		order_by = gdp
		max = 15
		check_range_bounds = no
		
		add_to_global_variable_list = {
			name = gdp_ranking_global_list
			target = this
		}

		change_local_variable = {
			name = gdp_local_var
			add = this.gdp
		}
		set_variable = { 
			name = gdp_value_tt
			value = this.gdp
		}
		change_local_variable = { 
			name = gdp_local_var
			divide = global_var:top_gdp_country_gvar
		}
		change_local_variable = { 
			name = gdp_local_var
			multiply = 880
		}
		set_variable = {
			name = gdp_var
			value = local_var:gdp_local_var
		}
	}
}
